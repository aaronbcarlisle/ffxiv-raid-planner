```ts
import { useMemo } from 'react';
import type { TierSnapshot, SnapshotPlayer } from '../types';
import { sortPlayersByRole } from '../utils/calculations';
import { SORT_PRESETS, DEFAULT_SETTINGS } from '../utils/constants';

export function useSortedMainRosterPlayers(tier: TierSnapshot | null): SnapshotPlayer[] {
  return useMemo(() => {
    const players = tier?.players ?? [];
    const preset = DEFAULT_SETTINGS.sortPreset ?? 'role';
    const order = SORT_PRESETS[preset]?.order ?? DEFAULT_SETTINGS.displayOrder;
    const sorted = sortPlayersByRole(players, order, preset);
    return sorted.filter((p) => p.configured && !p.isSubstitute);
  }, [tier]);
}
```

import { StaticSettingsHost } from '../components/settings';
import { useGroupAddToRoster } from './groupActionsContext';
import { useStaticGroupStore } from '../stores/staticGroupStore';
import { useCurrentTier } from '../stores/tierStore';

export function V2SettingsHost() {

      players={players}
      tierId={tier?.tierId}
      isAdmin={user?.isAdmin ?? false}
      onAddToRoster={onAddToRoster}
    />
  );
}

```ts
import { useMemo } from 'react';
import type { LootLogEntry, PageLedgerEntry } from '../types';
import { getTierById } from '../gamedata/raid-tiers';

export interface FightLootSummary { fight: string; cleared: boolean; dropCount: number; }

export function computeWeeklyLootSummary(
  floors: string[], lootLog: LootLogEntry[], pageLedger: PageLedgerEntry[], week: number,
): FightLootSummary[] {
  const weekLoot = lootLog.filter((e) => e.weekNumber === week);
  const clearedFloors = new Set(
    pageLedger.filter((e) => e.weekNumber === week && e.transactionType === 'earned').map((e) => e.floor),
  );
  return floors.map((fight) => ({
    fight,
    cleared: clearedFloors.has(fight),
    dropCount: weekLoot.filter((e) => e.floor === fight).length,
  }));
}

export function useWeeklyLootSummary(
  { tierId, lootLog, pageLedger, week }: { tierId?: string; lootLog: LootLogEntry[]; pageLedger: PageLedgerEntry[]; week: number },
): FightLootSummary[] {
  const floors = getTierById(tierId)?.floors ?? [];
  return useMemo(() => computeWeeklyLootSummary(floors, lootLog, pageLedger, week),
    [floors, lootLog, pageLedger, week]);
}
```
